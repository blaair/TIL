
```
// fs 모듈을 이용해 파일 전체를 읽어온다.
// toString() : 문자열 형태로 바꿔준다.
// split('\n') : 관례적으로 붙여주는 것. 입력을 받을 때 여러줄에 걸쳐서 들어올 때가 많아 줄바꿈 기준으로 나눔. 각각의 줄을 걸쳐서 라인을 읽어온다.
let fs = require('fs');
let input = fs.readFileSync('/dev/stdin').toString().split('\n');

// ["1", "2"]
let line = input[0].split(' '); 
let a = parseInt(line[0]); // 1
let b = parseInt(line[1]); // 2 

console.log(a + b);
```
